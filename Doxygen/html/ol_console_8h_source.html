74 Usage:
75 ~~~~~~
76 This class is abstract, so you must inherit from it. Override the OnUserCreate() function
77 with all the stuff you need for your application (for thready reasons it's best to do
78 this in this function and not your class constructor). Override the OnUserUpdate(float fElapsedTime)
79 function with the good stuff, it gives you the elapsed time since the last call so you
80 can modify your stuff dynamically. Both functions should return true, unless you need
81 the application to close.

294  void Start()
295  {
296  // Start the thread
297  m_bAtomActive = true;
298  std::thread t = std::thread(&olcConsoleGameEngine::GameThread, this);
299 
300  // Wait for thread to be exited
301  t.join();
302  }
303 private:
304  void GameThread()
305  {
306  // Create user resources as part of this thread
307  if (!OnUserCreate())
308  m_bAtomActive = false;
309 
310  auto tp1 = std::chrono::system_clock::now();
311  auto tp2 = std::chrono::system_clock::now();
312 
313  while (m_bAtomActive)
314  {
315  // Run as fast as possible
316  while (m_bAtomActive)
317  {
318  // Handle Timing
319  tp2 = std::chrono::system_clock::now();
320  std::chrono::duration<float> elapsedTime = tp2 - tp1;
321  tp1 = tp2;
322  float fElapsedTime = elapsedTime.count();
323 
324  // Handle Keyboard Input
325  for (int i = 0; i < 256; i++)
326  {
327  m_keyNewState[i] = GetAsyncKeyState(i);
328 
329  m_keys[i].bPressed = false;
330  m_keys[i].bReleased = false;
331 
332  if (m_keyNewState[i] != m_keyOldState[i])
333  {
334  if (m_keyNewState[i] & 0x8000)
335  {
336  m_keys[i].bPressed = !m_keys[i].bHeld;
337  m_keys[i].bHeld = true;
338  }
339  else
340  {
341  m_keys[i].bReleased = true;
342  m_keys[i].bHeld = false;
343  }
344  }
345 
346  m_keyOldState[i] = m_keyNewState[i];
347  }
348 
349  // Handle Mouse Input - Check for window events
350  INPUT_RECORD inBuf[32];
351  DWORD events = 0;
352  GetNumberOfConsoleInputEvents(m_hConsoleIn, &events);
353  if (events > 0)
354  ReadConsoleInput(m_hConsoleIn, inBuf, events, &events);
355 
356  // Handle events - we only care about mouse clicks and movement
357  // for now
358  for (DWORD i = 0; i < events; i++)
359  {
360  switch (inBuf[i].EventType)
361  {
362  case FOCUS_EVENT:
363  {
364  m_bConsoleInFocus = inBuf[i].Event.FocusEvent.bSetFocus;
365  }
366  break;
367 
368  case MOUSE_EVENT:
369  {
370  switch (inBuf[i].Event.MouseEvent.dwEventFlags)
371  {
372  case MOUSE_MOVED:
373  {
374  m_mousePosX = inBuf[i].Event.MouseEvent.dwMousePosition.X;
375  m_mousePosY = inBuf[i].Event.MouseEvent.dwMousePosition.Y;
376  }
377  break;
378 
379  case 0:
380  {
381  for (int m = 0; m < 5; m++)
382  m_mouseNewState[m] = (inBuf[i].Event.MouseEvent.dwButtonState & (1 << m)) > 0;
383 
384  }
385  break;
386 
387  default:
388  break;
389  }
390  }
391  break;
392 
393  default:
394  break;
395  // We don't care just at the moment
396  }
397  }
398 
399  for (int m = 0; m < 5; m++)
400  {
401  m_mouse[m].bPressed = false;
402  m_mouse[m].bReleased = false;
403 
404  if (m_mouseNewState[m] != m_mouseOldState[m])
405  {
406  if (m_mouseNewState[m])
407  {
408  m_mouse[m].bPressed = true;
409  m_mouse[m].bHeld = true;
410  }
411  else
412  {
413  m_mouse[m].bReleased = true;
414  m_mouse[m].bHeld = false;
415  }
416  }
417 
418  m_mouseOldState[m] = m_mouseNewState[m];
419  }
420 
421 
422  // Handle Frame Update
423  if (!OnUserUpdate(fElapsedTime))
424  m_bAtomActive = false;
425 
426  // Update Title & Present Screen Buffer
427  wchar_t s[256];
428  swprintf_s(s, 256, L"OneLoneCoder.com - Console Game Engine - %s - FPS: %3.2f", m_sAppName.c_str(), 1.0f / fElapsedTime);
429  SetConsoleTitle(s);
430  }
431 
432  // Allow the user to free resources if they have overrided the destroy function
433  if (OnUserDestroy())
434  {
435  // User has permitted destroy, so exit and clean up
436  delete[] m_bufScreen;
437  SetConsoleActiveScreenBuffer(m_hOriginalConsole);
438  m_cvGameFinished.notify_one();
439  }
440  else
441  {
442  // User denied destroy for some reason, so continue running
443  m_bAtomActive = true;
444  }
445  }
446  }

501  static BOOL CloseHandler(DWORD evt)
502  {
503  // Note this gets called in a seperate OS thread, so it must
504  // only exit when the game has finished cleaning up, or else
505  // the process will be killed before OnUserDestroy() has finished
506  if (evt == CTRL_CLOSE_EVENT)
507  {
508  m_bAtomActive = false;
509 
510  // Wait for thread to be exited
511  std::unique_lock<std::mutex> ul(m_muxGame);
512  m_cvGameFinished.wait(ul);
513  }
514  return true;
515  }
